\begin{song}{title=\centering Boží Vlak\\\normalsize \vspace*{-0.3cm}}  %% sem se napíše jméno songu a autor
\moveright 6cm \vbox{      %Varianta č. 1  ---> Jeden sloupec zarovnaný na střed	

\sloka 
	To boží vlak již jede 
	
	a míří zrovna sem
	
	jej pevná ruka vede,
	
	vlak bouří celou zem.
	
\refren
	Ó bratři sem,

	rychle, bratři, sem
	
	pojďte rychle, bratři, sem,
	
	bratři, ještě místa je tu dost všem.
	
\sloka
	Vlak zatáčkou teď sjíždí,
	
	je slyšet píšťal zvuk,
	
	již rachot kol se blíží,
	
	je slyšet vozů hluk.
	
\refren

\sloka
	Vlak jediný je na cestě
	
	a žádný jiný vsled,
	
	je navždy ztracen každý,
	
	kdo zůstane teď zpět.
	
\refren

}
\setcounter{Slokočet}{0}
\end{song}
